function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Hoja 1"); // Obtenemos los datos del formulario (FormData) var fecha = new Date(); var nombre = e.parameter.cliente; var telefono = e.parameter.telefono; var metodo = e.parameter.metodo; var monto = e.parameter.monto; var referencia = e.parameter.referencia; // Insertamos la fila en la planilla sheet.appendRow([fecha, nombre, telefono, metodo, monto, referencia]); // Devolvemos una respuesta de éxito return ContentService.createTextOutput("Éxito").setMimeType(ContentService.MimeType.TEXT); }
